country_event = {
	id = latvia.8
	title = latvia.8.t
	desc = latvia.8.d
	picture = GFX_report_event_strelnieki
	
	is_triggered_only = yes
	
	option = { # Accept demands
		name = latvia.8.a
		ai_chance = {
			base = 10
			modifier = { # Don't accept if in a faction that can protect them
				factor = 0
				OR = {
					any_other_country = {
						is_major = yes
						is_in_faction_with = EST
					}
					LAT = { surrender_progress > 0.5 }
				}
			}
		}
		LAT = {
			country_event = { days = 5 id = latvia.9 }
			
		}
		
	}
	option = { # Refuse
		name = latvia.8.b
		ai_chance = {
			base = 0
			modifier = {
				add = 1
				OR = {
					is_historical_focus_on = no
					any_other_country = {
						is_major = yes
						is_in_faction_with = EST
					}
				}
			}
			modifier = {
				add = 5
				any_other_country = {
					is_major = yes
					has_guaranteed = EST
				}
			}
		}

		LAT = {
			country_event = { days = 5 id = latvia.10 }
			effect_tooltip = {
				create_wargoal = {
					type = annex_everything
					target = EST
				}	
			}
		}
	}
}
